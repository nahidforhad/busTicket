let clickedCount = 0; function handleButtonClick(event) { const button = event.target; // Check if the button is already green if (!button.classList.contains('bg-green-500')) { // Check if the maximum limit of green buttons (4) is reached if (clickedCount < 4) { // Change button color to green button.classList.add('bg-green-500'); // Increment clicked count clickedCount++; } else { // Display message or perform action if maximum limit reached alert('You can only select up to 4 buttons.'); } } else { // If button is already green, change it back to default color button.classList.remove('bg-green-500n'); // Decrement clicked count clickedCount--; } } // Add event listeners to buttons in the specified sections var sectionButtons = document.querySelectorAll('#A1, #B1, #A2, #B2, #A3, #B3, #A4, #B4'); sectionButtons.forEach(function(button) { button.addEventListener('click', handleButtonClick); });
